export const Fragment = ({modType}) => {
    return(
        <ul>
            <li>In the <b>Unity Toolbar</b>: Go to <code>Tools → Template Wizard</code>.</li>
            <li>Make sure the <code>inspector</code> window is shown (Template Wizard will show there).</li>
            <li>Enter a <code>new mod folder name</code>. Example: <code>YourModName</code></li>
            <li>Set the <code>Mod Type</code> to: <code>{modType || '<YOUR_MOD_TYPE>'}</code>.</li>
            <li>Fill in the desired <code>ItemInfoConfig</code> fields.</li>
            {modType === 'Skin' && (
                <>
                    <li>Fill in the <code>SkinInfo</code>:
                        <ul>
                            <li><code>skinName</code>: The name of the model in the Skin folder you created，**Needs to be prefixed with your prefix**.</li>
                            <li><code>Name</code>: The name displayed in the store.</li>
                            <li><code>Desc</code>:The description of the item displayed in the store.</li>
                            <li><code>MeshRoot</code>:Name of the root node of all meshes in the prefab.</li>
                        </ul>
                    </li>
                </>
            )}
            <li>Click the <code>Generate Template</code> button to generate the <code>{modType || '<YOUR_MOD_TYPE>'}</code> folder structure.</li>
        </ul>
    )
}

export default Fragment;
